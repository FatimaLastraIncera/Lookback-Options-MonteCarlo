15 *
16 * @param S0 Current spot price.
17 * @param Smin Minimum observed spot.
18 * @param r Risk-free rate.
19 * @param sigma Volatility.
20 * @param T Time to maturity.
21 * @return Exact call price.
22 */
23double lookback_call_exact(const double S0,
24 const double Smin,
25 const double r,
26 const double sigma,
27 const double T)
28{
29 // Precompute frequently used quantities
30 const double sqrtT = std::sqrt(T);
31 const double sig2 = sigma * sigma;
32 const double logSM = std::log(S0 / Smin);
33
34 // Auxiliary variables from the closed-form formula
35 const double a1 = (logSM + (r + 0.5 * sig2) * T) / (sigma * sqrtT);
36 const double a2 = a1 - sigma * sqrtT;
37 const double a3 = (logSM + (-r + 0.5 * sig2) * T) / (sigma * sqrtT);
38
39 // Drift adjustment term
40 const double Y1 = -2.0 * (r - 0.5 * sig2) * logSM / sig2;
41
42 // Discount factor and volatility coefficient
43 const double discount = std::exp(-r * T);
44 const double coeff = sig2 / (2.0 * r);
45
46 // Pricing formula
47 const double part1 = S0 * ( norm_cdf(a1) - coeff * norm_cdf(-a1) );
48 const double part2 = -Smin * discount * ( norm_cdf(a2) - coeff * std::exp(Y1) * norm_cdf(-a3) );
49
50 return part1 + part2;
51}
